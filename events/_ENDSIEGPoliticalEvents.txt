	id = endsieg_political.4
	title = endsieg_political.4.t
	desc = endsieg_political.4.d
	picture = GFX_report_event_gathering_protest

	trigger = {
		has_idea_with_trait = communism_revolutionary

			limit = {
				has_dlc = "Together for Victory"
			}

				NOT = { tag = RAJ }
				AND = {
					tag = RAJ
					is_puppet = no
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 365
		modifier = {
			factor = 0.5
			has_country_flag = communism_popular_movement
		}
	}

	option = {
		name = endsieg_political.4.a
		add_popularity = {
			ideology = stalinist
			popularity = 0.15
		}
		add_stability = -0.02
	}
}
