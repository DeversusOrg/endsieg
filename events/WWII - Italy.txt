# Mussolini Arrested
country_event = {
	id = WW2_italy.8
	title = WW2_italy.8.t
	desc = WW2_italy.8.d
	picture = GFX_report_event_mussolini_arrested

	fire_only_once = yes

	trigger = {
		NOT = { date > 1944.1.1 }
		tag = ITA
		OR = {
			AND = {
				ITA = { surrender_progress > 0.20 }
				is_in_1943_or_earlier = yes
			}
			AND = {
				has_global_flag = welcome_event_1943
				date > 1943.7.24
			}
		}
	}

	mean_time_to_happen = { days = 1 }

	immediate = {
		set_global_flag = mussolini_gone

	option = {
