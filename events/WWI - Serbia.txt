# Balkan League
country_event = {
	id = ww1_serbia.1
	title = ww1_serbia.1.t
	desc = ww1_serbia.1.d
	picture = GFX_report_event_balkan_league

	fire_only_once = yes

	trigger = {
		has_global_flag = welcome_event_1910
		NOT = { date > 1913.12.1 }
		tag = SER
		date > 1912.2.2
	}

	mean_time_to_happen = { days = 1 }

	option = {
		name = ww1_serbia.1.a #Send the invitations!
		SER = {
			create_faction = BALKAN_LEAGUE
			#add_to_faction = MN1
			MN1 = { country_event = { id = ww1_serbia.101 hours = 3 } }
			#add_to_faction = GR1
			GR1 = { country_event = { id = ww1_serbia.101 hours = 3 } }
			#add_to_faction = BNN
			BNN = { country_event = { id = ww1_serbia.101 hours = 3 } }
			#Romania gets the option to join as well, but will only send minimal help, although a player can join in as well.
			RMM = { country_event = { id = ww1_serbia.102 hours = 3 } }
		}
	}
}

# First Balkan War
country_event = {
	id = ww1_serbia.2
	title = ww1_serbia.2.t
	desc = ww1_serbia.2.d
	picture = GFX_report_event_first_balkan_war

	fire_only_once = yes

	trigger = {
		NOT = { date > 1913.12.1 }
		NOT = { has_global_flag = let_go_balkans }
		tag = SER
		date > 1912.10.7
		#has_global_flag = treaty_of_ouchy #The Italians are accounted for now.
	}

	mean_time_to_happen = { days = 1 }

	option = {
		name = ww1_serbia.2.a
		add_threat = 5

		every_country = {
			limit = { is_in_faction_with = SER }
			declare_war_on = { target = TUR type = annex_everything }
		}
		TUR = {
			add_timed_idea = { idea = surprise_attack days = 45 }
		}
		if = {
			limit = {
				IT1 = { has_war_with = TUR }
			}
		add_to_faction = BALKAN_LEAGUE
		}
		set_global_flag = first_balkan_war
	}
}

#Balkan War 1 - Peace?
country_event = {
	id = ww1_serbia.301
	title = ww1_serbia.301.t
	desc = ww1_serbia.301.d
	picture = GFX_report_event_balkan_league

	fire_only_once = yes

	trigger = {
		NOT = { has_global_flag = franz_dead }  #WW1 hasn't started.
		has_global_flag = welcome_event_1910
		has_global_flag = first_balkan_war
		has_war_with = TUR
		tag = SER

		TUR = {
			OR = {
				AND = {
					surrender_progress > 0.10
					is_ai = yes
					SER = { is_ai = yes }
				}
				AND = {
					NOT = {
						controls_state = 731 	#Salonika
						controls_state = 793 	#Central Epirus
						controls_state = 790 	#Northern Epirus
						controls_state = 44 	#Albania
						controls_state = 791 	#Outer Montenegro
						controls_state = 795 	#Eastern Rumelia no ban plz
						controls_state = 766 	#East Macedonia
						controls_state = 789 	#South Macedonia
						controls_state = 106 	#North Macedonia
						controls_state = 744 	#Kosovo
						controls_state = 788 	#Edirne
					}
				}
			}
		}
	}

	#Option only gets taken by ai if serbia is ai and a different country isnt
	option = {
		name = ww1_serbia.301.a

		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				any_country = {
					is_in_faction_with = SER
					is_ai = no
				}
			}
		}
		TUR = { country_event = { id =  ww1_serbia.302 hours = 6 } }
	}

	option = {
		name = ww1_serbia.301.b #How about we push for Constantinople instead?
		ai_chance = { factor = 0 }
		add_political_power = -150
		add_stability = -0.05
		add_war_support = -0.05
	}

	option = {
		name = ww1_serbia.301.w #AI Serbia passes peace decision on to Human Balkans. In MP, each have an equal chance.
		ai_chance = {
			factor = 0 #Because 100 is a neat number
			modifier = {
				any_country ={
					is_in_faction_with = SER
				}
				factor = 100
			}
		}
		trigger = {
			SER = { is_ai = yes }
			any_country = {
				is_in_faction_with = SER
				is_ai = no
			}
		}
		random_country = {
			limit = {
				is_in_faction_with = SER
				is_ai = no
			}
			country_event = { id = ww1_serbia.301 }
		}
	}
}
